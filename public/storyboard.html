import OpenAI from "openai";
import { autoval } from "autoval";

const client = new OpenAI();
autoval.instrument(client);

const res = await client.chat
  .completions.create({
    model: "gpt-4o",
    messages: [
      { role: "user", content: prompt }
    ],
  });

One Config File
Tell Autoval where your logs are, where your prompt lives, and where to submit PRs.
# autoval.config.yaml

sink:
  type: clickhouse        # or: postgres, supabase, bigquery, jsonl
  url: $CLICKHOUSE_URL
  table: llm_call_logs
  input_column: input
  output_column: output

prompt:
  file: prompts/system-prompt.txt

evals:
  runner: braintrust       # or: vitest, pytest, custom
  dir: evals/

github:
  token: $GITHUB_TOKEN
  repo: meta/journal-agent

slack:
  token: $SLACK_BOT_TOKEN

judge:
  model: gemini-2.5-flash
  grounding: nimble       # web search for evidence before scoring

nimble:
  api_key: $NIMBLE_API_KEY
